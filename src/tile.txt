type tile_type =
  | ATile
  | HTile
  | VTile
(* ATile are types of tiles that are 1x1. HTiles are types of tiles that are 1x2
   (horizontal). VTiles are types of tiles that are 2x1 (vertical). *)

type t
(** Type t repressents a tile, which has a string of 1 or 2 characters
    associated with it, a tile type of either ATile, BTile, or CTile, and a
    position (x,y) on a game grid, where x represents its position on the
    x-axis, and y represents its position on the y-axis.*)

val tile_list : t list
(* [tile_list] is a list of tiles with random, valid positions 
that don't overlap.*)
